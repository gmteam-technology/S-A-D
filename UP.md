# 1. Inicializar git (se necessário)
git init

# 2. Adicionar remoto (substitua pela URL do seu repositório)
git remote add origin https://github.com/SEU_USUARIO/SEU_REPOSITORIO.git

# 3. Adicionar arquivos
git add .

# 4. Commit
git commit -m "Initial commit: SIAD Agro - Sistema completo"

# 5. Definir branch principal
git branch -M main

# 6. Push
git push -u origin main
